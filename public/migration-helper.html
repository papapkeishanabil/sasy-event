🔄 Migrasi Invitation Tracking

Menambahkan kolom invitation_sent dan invitation_sent_time ke tabel guests

-- Safe Migration: Run this in Supabase SQL Editor DO $$ BEGIN IF NOT EXISTS ( SELECT 1 FROM information_schema.columns WHERE table_name = 'guests' AND column_name = 'invitation_sent' ) THEN ALTER TABLE guests ADD COLUMN invitation_sent BOOLEAN DEFAULT FALSE; END IF; END $$; DO $$ BEGIN IF NOT EXISTS ( SELECT 1 FROM information_schema.columns WHERE table_name = 'guests' AND column_name = 'invitation_sent_time' ) THEN ALTER TABLE guests ADD COLUMN invitation_sent_time TIMESTAMPTZ; END IF; END $$; -- Verify columns SELECT column_name, data_type, is_nullable FROM information_schema.columns WHERE table_name = 'guests' AND column_name IN ('invitation_sent', 'invitation_sent_time');
